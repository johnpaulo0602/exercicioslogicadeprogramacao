   // Seção de Comandos, procedimento, funções, operadores, etc...

   escreva("Informe a nota da VAIII: ")

      escreval("VALOR INCORRETO!!")
      escreval("Favor inserir um valor entre 0 e 35")
      leia(va3)
   fimenquanto
      escreva("Informe a nota da OAT: ")
   leia (oat)
   enquanto (oat<0)ou(oat>25) faca
      escreval("VALOR INCORRETO!!")
      escreval("Favor inserir um valor entre 0 e 25")
      leia(oat)
   fimenquanto
   soma:=va1+va2+va3+oat
   escreval("A soma das notas é: ",soma)
   se (soma<40) entao
      escreval("O aluno está REPROVADO, pois sua nota foi inferior a 40 pontos")
   senao
        se (soma>=40) e (soma<60) entao
           escreval ("O aluno está de RECUPERAÇÃO, pois sua nota foi inferior a 60 pontos e superior a 40 pontos")
        senao
             escreval("O aluno está APROVADO, pois sua nota foi superior a 60 pontos")
        fimse
   fimse

            Algoritmo "Ordem crescente"

            Var
               // Seção de Declarações das variáveis
               n1,n2,n3,menor,meio,maior: inteiro
            
            Inicio
               // Seção de Comandos, procedimento, funções, operadores, etc...
               escreva("Informe o primeiro número: ")
               leia (n1)
               escreva("Informe o segundo número: ")
               leia (n2)
               escreva("Informe o terceiro número: ")
               leia (n3)
               //Verificando qual é o menor número
               se (n1n2) e (n1>n3) entao
                  maior:= n1
               senao
                  se (n2>n1) e (n2>n3) entao
                     maior:= n2
                  senao
                     maior:=n3
                  fimse
               fimse
               //Descobrir o número do meio
               meio:= (n1+n2+n3) - maior - menor
            
               escreval("Os números em ordem crescente são: ",menor,meio,maior)
            
            Fimalgoritmo
