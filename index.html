Exercicio 04 - Aula 01


Algoritmo "Novo Salário"

Var
   // Seção de Declarações das variáveis

Inicio
   // Seção de Comandos, procedimento, funções, operadores, etc...
   escreva("Digite o seu salário: ")

Exercicio 05 - Aula 01


Algoritmo "Percentual de presença"

Var
   // Seção de Declarações das variáveis
   hraula, faltas, presenca, percentual: real

Inicio
   // Seção de Comandos, procedimento, funções, operadores, etc...
   escreva("Digite a quantidade de horas aula: ")
   leia (hraula)
   escreva("Digite a quantidade de faltas: ")
   leia (faltas)
   presenca:= hraula-faltas
   percentual:= (presenca/hraula)*100
   escreval("O percentual de faltas desse aluno é de: ",percentual,"%")
Fimalgoritmo
